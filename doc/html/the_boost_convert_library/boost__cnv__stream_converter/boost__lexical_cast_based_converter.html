boost::lexical_cast-Based Converter 

#include <boost/convert.hpp>
#include <boost/convert/lexical_cast.hpp>

using std::string;
using boost::convert;

struct boost::cnv::custom : public boost::cnv::lexical_cast {};


int    i = convert<int>("123").value();
string s = convert<string>(123).value();

BOOST_TEST(i == 123);
BOOST_TEST(s == "123");
